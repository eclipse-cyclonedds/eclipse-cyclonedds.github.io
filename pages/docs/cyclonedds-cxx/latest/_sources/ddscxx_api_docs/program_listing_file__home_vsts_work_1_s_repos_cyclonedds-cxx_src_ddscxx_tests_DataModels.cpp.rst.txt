   /*testing implicit defaults of structs*/

   {
     using namespace DataModels_testing;
   
     implicit_defaults_struct ds;

     EXPECT_EQ(ds.c(), '\0');

     EXPECT_EQ(ds.b(), 0);
   }

     explicit_defaults_struct ds;
