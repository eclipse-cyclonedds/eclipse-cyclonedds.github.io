Roundtrip¶

Measures the roundtrip duration when sending and receiving a single message.

Design¶

The Roundtrip example consists of two units:

  • ping: Sends a message to pong and waits for its return.

  • pong: Waits for messages from ping and sends the same message back.

Scenario¶

A message is sent by the ping executable on the “PING” partition, which the pong executable is waiting for. The pong executable sends the same message back on the “PONG” partition, which the ping executable is waiting for. This sequence is repeated a configurable number of times.

The ping executable measures:

  • writeAccess time: time the write() method took.

  • readAccess time: time the take() method took.

  • roundTrip time: time between the call to the write() method and the return of the take() method.

  • ping also calculates min/max/average statistics on these values over a configurable number of samples and/or time out period.

Configurable:

  • payloadSize: the size of the payload in bytes.

  • numSamples: the number of samples to send.

  • timeOut: the number of seconds ping should run for.

Running the example¶

To avoid mixing the output, run the ping and pong in separate terminals.

  1. Open two terminals.

  2. In the first terminal start Pong by running pong.

    pong usage:

    ./pong

  3. In the second terminal start Ping by running ping.

    ping usage (parameters must be supplied in order):

    ./ping [payloadSize (bytes, 0 - 655536)] [numSamples (0 = infinite)] [timeOut (seconds, 0 = infinite)]

    ./ping quit - ping sends a quit signal to pong.

    defaults:

    ./ping 0 0 0

  4. To achieve optimal performance, set the CPU affinity so that ping and pong run on separate CPU cores and use real-time scheduling:

Pong usage:

taskset -c 0 chrt -f 80 ./pong

Ping usage:

taskset -c 1 chrt -f 80 ./ping [payloadSize (bytes, 0 - 655536)] [numSamples (0 = infinite)] [timeOut (seconds, 0 = infinite)]

Pong usage:

START /affinity 1 /high cmd /k "pong.exe"

Ping usage:

START /affinity 2 /high cmd /k "ping.exe" [payloadSize (bytes, 0 - 655536)] [numSamples (0 = infinite)] [timeOut (seconds, 0 = infinite)]

 56static ExampleTimeStats *exampleAddTimingToTimeStats
 57  (ExampleTimeStats *stats, dds_time_t timing)
 58{
 59  if (stats->valuesSize > stats->valuesMax)
 60  {
 61    dds_time_t * temp = (dds_time_t*) realloc (stats->values, (stats->valuesMax + TIME_STATS_SIZE_INCREMENT) * sizeof (dds_time_t));
 62    stats->values = temp;
 63    stats->valuesMax += TIME_STATS_SIZE_INCREMENT;
 64  }
 65  if (stats->values != NULL && stats->valuesSize < stats->valuesMax)
 66  {
 67    stats->values[stats->valuesSize++] = timing;
 68  }
 69  stats->average = ((double)stats->count * stats->average + (double)timing) / (double)(stats->count + 1);
 70  stats->min = (stats->count == 0 || timing < stats->min) ? timing : stats->min;
 71  stats->max = (stats->count == 0 || timing > stats->max) ? timing : stats->max;
 72  stats->count++;
 73
 74  return stats;
 75}

 77static int exampleCompareul (const void* a, const void* b)
 78{
 79  dds_time_t ul_a = *((dds_time_t*)a);
 80  dds_time_t ul_b = *((dds_time_t*)b);
 81
 82  if (ul_a < ul_b) return -1;
 83  if (ul_a > ul_b) return 1;
 84  return 0;
 85}

 87static double exampleGetMedianFromTimeStats (ExampleTimeStats *stats)
 88{
 89  double median = 0.0;
 90
 91  qsort (stats->values, stats->valuesSize, sizeof (dds_time_t), exampleCompareul);
 92
 93  if (stats->valuesSize % 2 == 0)
 94  {
 95    median = (double)(stats->values[stats->valuesSize / 2 - 1] + stats->values[stats->valuesSize / 2]) / 2;
 96  }
 97  else
 98  {
 99    median = (double)stats->values[stats->valuesSize / 2];
100  }
101
102  return median;
103}

155static void data_available(dds_entity_t rd, void *arg)
156{
157  dds_time_t difference = 0;
158  int status;
159  (void)arg;
160  /* Take sample and check that it is valid */
161  preTakeTime = dds_time ();
162  status = dds_take (rd, samples, info, MAX_SAMPLES, MAX_SAMPLES);
163  if (status < 0)
164    DDS_FATAL("dds_take: %s\n", dds_strretcode(-status));
165  postTakeTime = dds_time ();
166
167  /* Update stats */
168  difference = (postWriteTime - preWriteTime)/DDS_NSECS_IN_USEC;
169  writeAccess = *exampleAddTimingToTimeStats (&writeAccess, difference);
170  writeAccessOverall = *exampleAddTimingToTimeStats (&writeAccessOverall, difference);
171
172  difference = (postTakeTime - preTakeTime)/DDS_NSECS_IN_USEC;
173  readAccess = *exampleAddTimingToTimeStats (&readAccess, difference);
174  readAccessOverall = *exampleAddTimingToTimeStats (&readAccessOverall, difference);
175
176  difference = (postTakeTime - info[0].source_timestamp)/DDS_NSECS_IN_USEC;
177  roundTrip = *exampleAddTimingToTimeStats (&roundTrip, difference);
178  roundTripOverall = *exampleAddTimingToTimeStats (&roundTripOverall, difference);
179
180  if (!warmUp) {
181    /* Print stats each second */
182    difference = (postTakeTime - startTime)/DDS_NSECS_IN_USEC;
183    if (difference > US_IN_ONE_SEC)
184    {
185      printf("%9" PRIi64 " %9lu %8.0f %8" PRIi64 " %8" PRIi64 " %8" PRIi64 " %10lu %8.0f %8" PRIi64 " %10lu %8.0f %8" PRIi64 "\n",
186             elapsed + 1,
187             roundTrip.count,
188             exampleGetMedianFromTimeStats (&roundTrip) / 2,
189             roundTrip.min / 2,
190             exampleGet99PercentileFromTimeStats (&roundTrip) / 2,
191             roundTrip.max / 2,
192             writeAccess.count,
193             exampleGetMedianFromTimeStats (&writeAccess),
194             writeAccess.min,
195             readAccess.count,
196             exampleGetMedianFromTimeStats (&readAccess),
197             readAccess.min);
198      fflush (stdout);
199
200      exampleResetTimeStats (&roundTrip);
201      exampleResetTimeStats (&writeAccess);
202      exampleResetTimeStats (&readAccess);
203      startTime = dds_time ();
204      elapsed++;
205    }
206  }
207
208  preWriteTime = dds_time();
209  status = dds_write_ts (writer, &pub_data, preWriteTime);
210  if (status < 0)
211    DDS_FATAL("dds_write_ts: %s\n", dds_strretcode(-status));
212  postWriteTime = dds_time();
213}

441static dds_entity_t prepare_dds(dds_entity_t *wr, dds_entity_t *rd, dds_entity_t *rdcond, dds_listener_t *listener)
442{
443  dds_return_t status;
444  dds_entity_t topic;
445  dds_entity_t publisher;
446  dds_entity_t subscriber;
447
448  const char *pubPartitions[] = { "ping" };
449  const char *subPartitions[] = { "pong" };
450  dds_qos_t *pubQos;
451  dds_qos_t *dwQos;
452  dds_qos_t *drQos;
453  dds_qos_t *subQos;
454
455  /* A DDS_Topic is created for our sample type on the domain participant. */
456  topic = dds_create_topic (participant, &RoundTripModule_DataType_desc, "RoundTrip", NULL, NULL);
457  if (topic < 0)
458    DDS_FATAL("dds_create_topic: %s\n", dds_strretcode(-topic));
459
460  /* A DDS_Publisher is created on the domain participant. */
461  pubQos = dds_create_qos ();
462  dds_qset_partition (pubQos, 1, pubPartitions);
463
464  publisher = dds_create_publisher (participant, pubQos, NULL);
465  if (publisher < 0)
466    DDS_FATAL("dds_create_publisher: %s\n", dds_strretcode(-publisher));
467  dds_delete_qos (pubQos);
468
469  /* A DDS_DataWriter is created on the Publisher & Topic with a modified Qos. */
470  dwQos = dds_create_qos ();
471  dds_qset_reliability (dwQos, DDS_RELIABILITY_RELIABLE, DDS_SECS (10));
472  dds_qset_writer_data_lifecycle (dwQos, false);
473  *wr = dds_create_writer (publisher, topic, dwQos, NULL);
474  if (*wr < 0)
475    DDS_FATAL("dds_create_writer: %s\n", dds_strretcode(-*wr));
476  dds_delete_qos (dwQos);
477
478  /* A DDS_Subscriber is created on the domain participant. */
479  subQos = dds_create_qos ();
480
481  dds_qset_partition (subQos, 1, subPartitions);
482
483  subscriber = dds_create_subscriber (participant, subQos, NULL);
484  if (subscriber < 0)
485    DDS_FATAL("dds_create_subscriber: %s\n", dds_strretcode(-subscriber));
486  dds_delete_qos (subQos);
487  /* A DDS_DataReader is created on the Subscriber & Topic with a modified QoS. */
488  drQos = dds_create_qos ();
489  dds_qset_reliability (drQos, DDS_RELIABILITY_RELIABLE, DDS_SECS(10));
490  *rd = dds_create_reader (subscriber, topic, drQos, listener);
491  if (*rd < 0)
492    DDS_FATAL("dds_create_reader: %s\n", dds_strretcode(-*rd));
493  dds_delete_qos (drQos);
494
495  waitSet = dds_create_waitset (participant);
496  if (listener == NULL) {
497    *rdcond = dds_create_readcondition (*rd, DDS_ANY_STATE);
498    status = dds_waitset_attach (waitSet, *rdcond, *rd);
499    if (status < 0)
500      DDS_FATAL("dds_waitset_attach: %s\n", dds_strretcode(-status));
501  } else {
502    *rdcond = 0;
503  }
504  status = dds_waitset_attach (waitSet, waitSet, waitSet);
505  if (status < 0)
506    DDS_FATAL("dds_waitset_attach: %s\n", dds_strretcode(-status));
507
508  return participant;
509}

 42static void data_available(dds_entity_t rd, void *arg)
 43{
 44  int status, samplecount;
 45  (void)arg;
 46  samplecount = dds_take (rd, samples, info, MAX_SAMPLES, MAX_SAMPLES);
 47  if (samplecount < 0)
 48    DDS_FATAL("dds_take: %s\n", dds_strretcode(-samplecount));
 49  for (int j = 0; !dds_triggered (waitSet) && j < samplecount; j++)
 50  {
 51    /* If writer has been disposed terminate pong */
 52
 53    if (info[j].instance_state == DDS_IST_NOT_ALIVE_DISPOSED)
 54    {
 55      printf ("Received termination request. Terminating.\n");
 56      dds_waitset_set_trigger (waitSet, true);
 57      break;
 58    }
 59    else if (info[j].valid_data)
 60    {
 61      /* If sample is valid, send it back to ping */
 62      RoundTripModule_DataType * valid_sample = &data[j];
 63      status = dds_write_ts (writer, valid_sample, info[j].source_timestamp);
 64      if (status < 0)
 65        DDS_FATAL("dds_write_ts: %d\n", -status);
 66    }
 67  }
 68}
 69
 70int main (int argc, char *argv[])
 71{
 72  dds_duration_t waitTimeout = DDS_INFINITY;
 73  unsigned int i;
 74  int status;
 75  dds_attach_t wsresults[1];
 76  size_t wsresultsize = 1U;
 77
 78  dds_listener_t *listener = NULL;
 79  bool use_listener = false;
 80  int argidx = 1;
 81
 82  if (argc > argidx && strcmp(argv[argidx], "-l") == 0)
 83  {
 84    argidx++;
 85    use_listener = true;
 86  }
 87
 88  /* Register handler for Ctrl-C */
 89
 90#ifdef _WIN32
 91  DDSRT_WARNING_GNUC_OFF(cast-function-type)
 92  SetConsoleCtrlHandler ((PHANDLER_ROUTINE)CtrlHandler, TRUE);
 93  DDSRT_WARNING_GNUC_ON(cast-function-type)
 94#elif !DDSRT_WITH_FREERTOS
 95  struct sigaction sat, oldAction;
 96  sat.sa_handler = CtrlHandler;
 97  sigemptyset (&sat.sa_mask);
 98  sat.sa_flags = 0;
 99  sigaction (SIGINT, &sat, &oldAction);
100#endif
101
102  /* Initialize sample data */
103  memset (data, 0, sizeof (data));
104  for (i = 0; i < MAX_SAMPLES; i++)
105  {
106    samples[i] = &data[i];
107  }
108
109  participant = dds_create_participant (DDS_DOMAIN_DEFAULT, NULL, NULL);
110  if (participant < 0)
111    DDS_FATAL("dds_create_participant: %s\n", dds_strretcode(-participant));
112
113  if (use_listener)
114  {
115    listener = dds_create_listener(NULL);
116    dds_lset_data_available(listener, data_available);
117  }
118
119  (void)prepare_dds(&writer, &reader, &readCond, listener);
120
121  while (!dds_triggered (waitSet))
122  {
123    /* Wait for a sample from ping */
124
125    status = dds_waitset_wait (waitSet, wsresults, wsresultsize, waitTimeout);
126    if (status < 0)
127      DDS_FATAL("dds_waitset_wait: %s\n", dds_strretcode(-status));
128
129    /* Take samples */
130    if (listener == NULL) {
131      data_available (reader, 0);
132    }
133  }
134
135#ifdef _WIN32
136  SetConsoleCtrlHandler (0, FALSE);
137#elif !DDSRT_WITH_FREERTOS
138  sigaction (SIGINT, &oldAction, 0);
139#endif
140
141  /* Clean up */
142  finalize_dds(participant, data);
143
144  return EXIT_SUCCESS;
145}
146
147static void finalize_dds(dds_entity_t pp, RoundTripModule_DataType xs[MAX_SAMPLES])
148{
149  dds_return_t status;
150  status = dds_delete (pp);
151  if (status < 0)
152    DDS_FATAL("dds_delete: %s\n", dds_strretcode(-status));
153  for (unsigned int i = 0; i < MAX_SAMPLES; i++)
154  {
155    RoundTripModule_DataType_free (&xs[i], DDS_FREE_CONTENTS);
156  }
157}

159static dds_entity_t prepare_dds(dds_entity_t *wr, dds_entity_t *rd, dds_entity_t *rdcond, dds_listener_t *rdlist)
160{
161  const char *pubPartitions[] = { "pong" };
162  const char *subPartitions[] = { "ping" };
163  dds_qos_t *qos;
164  dds_entity_t subscriber;
165  dds_entity_t publisher;
166  dds_entity_t topic;
167  dds_return_t status;
168
169  /* A DDS Topic is created for our sample type on the domain participant. */
170
171  topic = dds_create_topic (participant, &RoundTripModule_DataType_desc, "RoundTrip", NULL, NULL);
172  if (topic < 0)
173    DDS_FATAL("dds_create_topic: %s\n", dds_strretcode(-topic));
174
175  /* A DDS Publisher is created on the domain participant. */
176
177  qos = dds_create_qos ();
178  dds_qset_partition (qos, 1, pubPartitions);
179
180  publisher = dds_create_publisher (participant, qos, NULL);
181  if (publisher < 0)
182    DDS_FATAL("dds_create_publisher: %s\n", dds_strretcode(-publisher));
183  dds_delete_qos (qos);
184
185  /* A DDS DataWriter is created on the Publisher & Topic with a modififed Qos. */
186
187  qos = dds_create_qos ();
188  dds_qset_reliability (qos, DDS_RELIABILITY_RELIABLE, DDS_SECS(10));
189  dds_qset_writer_data_lifecycle (qos, false);
190  *wr = dds_create_writer (publisher, topic, qos, NULL);
191  if (*wr < 0)
192    DDS_FATAL("dds_create_writer: %s\n", dds_strretcode(-*wr));
193  dds_delete_qos (qos);
194
195  /* A DDS Subscriber is created on the domain participant. */
196
197  qos = dds_create_qos ();
198  dds_qset_partition (qos, 1, subPartitions);
199
200  subscriber = dds_create_subscriber (participant, qos, NULL);
201  if (subscriber < 0)
202    DDS_FATAL("dds_create_subscriber: %s\n", dds_strretcode(-subscriber));
203  dds_delete_qos (qos);
204
205  /* A DDS DataReader is created on the Subscriber & Topic with a modified QoS. */
206
207  qos = dds_create_qos ();
208  dds_qset_reliability (qos, DDS_RELIABILITY_RELIABLE, DDS_SECS(10));
209  *rd = dds_create_reader (subscriber, topic, qos, rdlist);
210  if (*rd < 0)
211    DDS_FATAL("dds_create_reader: %s\n", dds_strretcode(-*rd));
212  dds_delete_qos (qos);
213
214  waitSet = dds_create_waitset (participant);
215  if (rdlist == NULL) {
216    *rdcond = dds_create_readcondition (*rd, DDS_ANY_STATE);
217    status = dds_waitset_attach (waitSet, *rdcond, *rd);
218    if (status < 0)
219      DDS_FATAL("dds_waitset_attach: %s\n", dds_strretcode(-status));
220  } else {
221    *rdcond = 0;
222  }
223  status = dds_waitset_attach (waitSet, waitSet, waitSet);
224  if (status < 0)
225    DDS_FATAL("dds_waitset_attach: %s\n", dds_strretcode(-status));
226
227  printf ("Waiting for samples from ping to send back...\n");
228  fflush (stdout);
229
230  return participant;
231}
